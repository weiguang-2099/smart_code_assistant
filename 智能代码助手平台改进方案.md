## 🎯 核心改进策略

### 1. 引入 LangGraph Agent 架构（核心亮点）

#### 为什么要用 Agent 模式？

**传统 LLM 调用的问题**：
```python
# ❌ 传统方式：直接调用 LLM
def generate_code(prompt):
    response = openai.chat.completions.create(
        model="gpt-4",
        messages=[{"role": "user", "content": prompt}]
    )
    return response.choices[0].message.content
```
**问题**：
- 无法处理复杂任务（需要多步骤推理）
- 无法调用外部工具（数据库查询、代码分析）
- 缺乏自我纠错能力

**Agent 模式的优势**：
```python
# ✅ Agent 模式：ReAct (Reasoning + Action)
agent = create_react_agent(llm, tools=[
    analyze_code_tool,      # 代码分析工具
    search_docs_tool,       # 文档搜索工具
    execute_code_tool,      # 代码执行工具
    review_code_tool,       # 代码审查工具
])
```

**工作流程**：
```
用户请求
   ↓
Agent 分析任务 (Reasoning)
   ↓
选择合适的 Tool (Action)
   ↓
执行 Tool 获取结果
   ↓
继续推理或返回结果
```

# app/core/agents/code_generator.py
from langchain_openai import ChatOpenAI
from langchain_core.tools import tool
from langgraph.prebuilt import create_react_agent
from pydantic import BaseModel, Field

# ============ Tool 定义 ============

class CodeAnalysisInput(BaseModel):
    """代码分析工具输入"""
    code: str = Field(description="要分析的代码")
    language: str = Field(default="python", description="编程语言")

@tool(args_schema=CodeAnalysisInput)
def analyze_code_structure(code: str, language: str = "python") -> str:
    """
    分析代码结构，提取函数、类、依赖等信息
    用于理解现有代码库，辅助代码生成
    """
    # 这里可以集成 tree-sitter 或 AST 解析
    return f"""分析语言 {language} 的代码结构：
    - 提取所有函数和类定义
    - 分析依赖关系
    - 识别代码模式
    代码：{code[:200]}..."""

class RepositorySearchInput(BaseModel):
    """仓库搜索工具输入"""
    query: str = Field(description="搜索关键词")
    file_type: str = Field(default="*", description="文件类型")

@tool(args_schema=RepositorySearchInput)
def search_code_repository(query: str, file_type: str = "*") -> str:
    """
    在代码仓库中搜索相关代码示例
    帮助 Agent 找到类似的实现参考
    """
    # 集成 Elasticsearch 或向量数据库
    return f"""搜索 "{query}" 相关的 {file_type} 文件：
    - 找到 5 个相似实现
    - 提取代码片段
    - 返回最佳实践"""

class CodeExecutionInput(BaseModel):
    """代码执行工具输入"""
    code: str = Field(description="要执行的代码")
    language: str = Field(default="python", description="编程语言")

@tool(args_schema=CodeExecutionInput)
def execute_code_sandbox(code: str, language: str = "python") -> str:
    """
    在沙箱环境中执行代码，验证生成的代码是否可运行
    支持 Python, JavaScript, C# 等语言
    """
    # 集成 Docker 沙箱或 AWS Lambda
    return f"""执行 {language} 代码：
    状态：成功
    输出：Hello World
    错误：无"""

class DocumentationSearchInput(BaseModel):
    """文档搜索工具输入"""
    library: str = Field(description="库或框架名称")
    topic: str = Field(description="搜索主题")

@tool(args_schema=DocumentationSearchInput)
def query_documentation(library: str, topic: str) -> str:
    """
    查询官方文档和最佳实践
    用于生成符合规范的代码
    """
    # 集成 RAG (检索增强生成)
    return f"""查询 {library} 关于 {topic} 的文档：
    - API 参考
    - 使用示例
    - 最佳实践"""

# ============ Agent 服务 ============

class CodeGeneratorAgent:
    """代码生成 Agent 服务"""

    def __init__(self, api_key: str, model: str = "gpt-4"):
        self.llm = ChatOpenAI(
            model=model,
            api_key=api_key,
            temperature=0.7,  # 稍高的温度保持创造性
        )

        self.tools = [
            analyze_code_structure,
            search_code_repository,
            execute_code_sandbox,
            query_documentation,
        ]

        self.agent = create_react_agent(
            self.llm,
            self.tools,
        )

    def _get_system_prompt(self) -> str:
        """系统提示"""
        return """你是一个专业的代码生成助手。你可以：
1. 使用 analyze_code_structure 分析现有代码
2. 使用 search_code_repository 搜索相似实现
3. 使用 execute_code_sandbox 验证生成的代码
4. 使用 query_documentation 查询文档

生成代码时请遵循：
- 代码规范和最佳实践
- 添加必要的注释
- 考虑边界条件和错误处理
- 确保代码可执行且无明显 bug"""

    async def generate_code(
        self,
        requirement: str,
        language: str = "python",
        context: str = "",
    ) -> dict:
        """
        生成代码

        Args:
            requirement: 需求描述
            language: 编程语言
            context: 上下文代码

        Returns:
            {
                "code": "生成的代码",
                "explanation": "代码解释",
                "steps": ["执行步骤1", "执行步骤2"]
            }
        """
        from langchain_core.messages import SystemMessage, HumanMessage

        user_message = f"""请生成 {language} 代码完成以下需求：

需求：{requirement}

上下文代码（如果有）：
{context}

请先分析需求，搜索相关示例，然后生成代码并验证。"""

        result = await self.agent.ainvoke({
            "messages": [
                SystemMessage(content=self._get_system_prompt()),
                HumanMessage(content=user_message)
            ]
        })

        # 提取最终代码
        messages = result.get("messages", [])
        final_response = messages[-1].content if messages else ""

        return {
            "code": self._extract_code(final_response),
            "explanation": final_response,
            "steps": self._extract_steps(result)
        }

    def _extract_code(self, response: str) -> str:
        """从响应中提取代码块"""
        import re
        code_blocks = re.findall(r'```[\w]*\n(.*?)```', response, re.DOTALL)
        return code_blocks[0] if code_blocks else response

    def _extract_steps(self, result: dict) -> list:
        """提取 Agent 执行步骤（用于调试和展示）"""
        messages = result.get("messages", [])
        steps = []
        for msg in messages:
            if hasattr(msg, 'tool_calls'):
                for tool_call in msg.tool_calls:
                    steps.append(f"调用工具: {tool_call['name']}")
        return steps

    ✅ 低危：无明显 XSS 风险"""

class CodeSmellInput(BaseModel):
    """代码坏味道检测输入"""
    code: str = Field(description="要检测的代码")

@tool(args_schema=CodeSmellInput)
def detect_code_smells(code: str) -> str:
    """
    检测代码坏味道
    - 重复代码
    - 过长函数
    - 过多参数
    - 神秘命名
    """
    return """代码坏味道检测：

    🟡 参数过多：process() 有 8 个参数
    🟡 命名不清：变量 x, y, z 意义不明"""

class CodeReviewerAgent:
    """代码审查 Agent"""

    def __init__(self, api_key: str):
        from langchain_openai import ChatOpenAI
        from langgraph.prebuilt import create_react_agent

        self.llm = ChatOpenAI(model="gpt-4", api_key=api_key)
        self.tools = [
            run_static_analysis,
            security_vulnerability_scan,
            detect_code_smells,
        ]
        self.agent = create_react_agent(self.llm, self.tools)

        """执行代码审查"""

// Services/AIServiceClient.cs
public class AIServiceClient
{
    private readonly HttpClient _httpClient;
    private readonly ILogger<AIServiceClient> _logger;

    public AIServiceClient(HttpClient httpClient, ILogger<AIServiceClient> logger)
    {
        _httpClient = httpClient;
        _logger = logger;
    }

    // 代码生成请求
    public async Task<CodeGenerationResponse> GenerateCodeAsync(
        CodeGenerationRequest request,
        CancellationToken ct = default)
    {
        var response = await _httpClient.PostAsJsonAsync(
            "/api/v1/code/generate",
            request,
            ct
        );

        response.EnsureSuccessStatusCode();
        return await response.Content.ReadFromJsonAsync<CodeGenerationResponse>(ct);
    }

    // 代码审查请求
    public async Task<CodeReviewResponse> ReviewCodeAsync(
        string code,
        string language = "python")
    {
        var request = new { code, language };
        var response = await _httpClient.PostAsJsonAsync(
            "/api/v1/review/analyze",
            request
        );

        return await response.Content.ReadFromJsonAsync<CodeReviewResponse>();
    }
}

using Microsoft.AspNetCore.SignalR;

public class CodeCollaborationHub : Hub
{
    private readonly AIServiceClient _aiService;

    public CodeCollaborationHub(AIServiceClient aiService)
    {
        _aiService = aiService;
    }

    // 用户加入代码编辑房间
    public async Task JoinRoom(string roomId)
    {
        await Groups.AddToGroupAsync(Context.ConnectionId, roomId);
        await Clients.Group(roomId).SendAsync("UserJoined", Context.User.Identity.Name);
    }

    // 实时代码同步
    public async Task SyncCode(string roomId, string code, int cursorPosition)
    {
        await Clients.OthersInGroup(roomId).SendAsync("CodeUpdated", new
        {
            Code = code,
            CursorPosition = cursorPosition,
            UserId = Context.User.Identity.Name,
            Timestamp = DateTime.UtcNow
        });
    }

    // 请求 AI 生成代码（实时流式返回）
    public async IAsyncEnumerable<string> StreamCodeGeneration(
        string requirement,
        string language)
    {
        // 调用 FastAPI 的流式接口
        var response = await _aiService.GenerateCodeAsync(new CodeGenerationRequest
        {
            Requirement = requirement,
            Language = language
        });

        // 逐步返回生成的代码
        foreach (var chunk in response.Code.Split('\n'))
        {
            yield return chunk;
            await Task.Delay(50); // 模拟打字机效果
        }
    }
}

CREATE TABLE code_tasks (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,

    input_data TEXT NOT NULL,

    INDEX idx_project_task (project_id, task_type),

);

-- 代码审查结果表
CREATE TABLE code_reviews (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    task_id BIGINT NOT NULL,
    code_hash VARCHAR(64) NOT NULL,  -- SHA256 哈希避免重复审查
    score INT NOT NULL,
    issues JSON NOT NULL,  -- [{level, message, line}]
    suggestions TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uk_code_hash (code_hash),
    FOREIGN KEY (task_id) REFERENCES code_tasks(id)
);

-- Agent 工具使用日志（用于分析和优化）
CREATE TABLE agent_tool_logs (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,

    tool_name VARCHAR(100) NOT NULL,
    input_params JSON,
    output_result TEXT,
    execution_time_ms INT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_task_tool (task_id, tool_name),

      .build();

    await connection.start();

    // 订阅流式返回
    connection.stream('StreamCodeGeneration', requirement, 'python')
      .subscribe({
        next: (chunk) => {
          setGeneratedCode(prev => prev + chunk + '\n');
        },
        error: (err) => console.error(err),
        complete: () => {
          setIsGenerating(false);
          connection.stop();
        }
      });
  };

  return (
    <div className="grid grid-cols-2 gap-4">
      {/* 需求输入 */}
      <div>
        <textarea
          value={requirement}
          onChange={(e) => setRequirement(e.target.value)}
          placeholder="描述你的需求..."
          className="w-full h-40 p-4 border rounded"
        />
        <button
          onClick={generateCode}
          disabled={isGenerating}
          className="mt-4 px-6 py-2 bg-blue-600 text-white rounded"
        >
          {isGenerating ? '生成中...' : '生成代码'}
        </button>

        {/* Agent 执行步骤 */}
        {steps.length > 0 && (
          <div className="mt-4 p-4 bg-gray-100 rounded">
            <h3 className="font-bold">AI 执行步骤：</h3>
            <ul className="list-disc pl-5">
              {steps.map((step, i) => (
                <li key={i}>{step}</li>
              ))}
            </ul>
          </div>
        )}
      </div>

      {/* 代码编辑器 */}
      <div>
        <Editor
          height="500px"
          language="python"
          value={generatedCode}
          onChange={(value) => setGeneratedCode(value || '')}
          theme="vs-dark"
        />
      </div>
    </div>
  );
}
